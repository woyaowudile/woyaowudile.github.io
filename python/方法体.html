# 方法


############-1-###############
def name(name):
	print(name)

name('zs')               # 'zs'

############-2-###############
def name(name = 'ls'):
	print(name)

name() # ls

############-3-###############
def name(*name):
	print(name)

name('zs', 'ls')         # ('zs', 'ls') 这种类型叫元祖，与list类似，但不能被修改， type类型：tuple

############-4-###############
def name(name):
	return name

############-5-###############
def name(name):
	return name,2,3

# return 可以返回多个结果，用逗号隔开
