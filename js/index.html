    // 将 数组对象 的键值， 重新赋值到 一个 规定的 键值组 中去
// 例如 from: [ {a:1,b:2}, {a:11,b22}] --> to: {a:''} ----> return:  [{a:1},{a:11}][0] --{a:1}
function changBean(from, to, isObj = true, index = 0) {
    /**
     *  from 为数组， to为对象, 
     *  默认返回值 是 对象，取值第一条。即： {a:1}
     *  如果需要返回的数组，可以 isObj = false
     */
    let o = [];
    from.map( (v, i) => {
        o[i] = {};
        for (let k in to) {
            o[i][k] = v[k];
        }
    });
    return isObj ? o[index] : o;
}
